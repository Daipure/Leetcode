slide window + 雙指針
左指針指向目前最小的
右指針往後移 當右指針大於左指針 代表有可能是最大

```cpp
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        // slide window + 雙指針
        // 左指針指向目前最小的
        // 右指針往後移 當右指針大於左指針 代表有可能是最大
        
        int l = 0;
        int r = 1;
        int max_profit = 0;
        int profit = 0;
        while(r<prices.size()){
            if(prices[l]<prices[r]){
                profit = prices[r] - prices[l];
                max_profit = max(max_profit,profit);
            }
            else{
                // 找到更小的
                l=r;
            }
            r++;
        }
        return max_profit;
    }
};
```
