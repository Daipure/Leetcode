雙指針的應用
nums[left]+nums[right]<target
代表固定left的情況 之間的right都會小於Target 所以right-left

```cpp
class Solution {
public:
    int countPairs(vector<int>& nums, int target) {
        // 雙指針
        sort(nums.begin(),nums.end());
        int count = 0;

        // -1 1 1 2 3  
        int left = 0; 
        int right = nums.size()-1;

        while(left<right){
            if(nums[left]+nums[right]<target){
                count+=right-left;
                left++;
            }
            else{
                right--;
            }
        }

        return count;
    }
};

```
