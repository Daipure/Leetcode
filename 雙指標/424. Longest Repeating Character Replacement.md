窗口內最長會出現在 頻率最多的元素中加上k個替換掉的
因此一個合法的窗口= 窗口長度-頻率最高的要<=k
右邊界向右移動 直到找不到不合法
使用 map

```cpp
class Solution {
public:
    int characterReplacement(string s, int k) {
        // 使用滑動窗口 對每個元素 往右移看該元素最大可以多長
        // 
        int n = s.size();
        if(n<2)return n;
        // 窗口內最長會出現在 頻率最多的元素中加上k個替換掉的
        // 因此一個合法的窗口= 窗口長度-頻率最高的要<=k
        // 右邊界向右移動 直到找不到不合法
        // 使用 map

        unordered_map<char, int> count;
        int max_freq=0;
        int res =  0;
        int l = 0;

        for(int r=0;r<n;r++){
            count[s[r]]++;    
            max_freq = max(count[s[r]],max_freq);

            // 非法
            while( (r-l+1) -max_freq >k ){
                count[s[l]]--;
                l++;
            }
            res = max(res,r-l+1);

        }
        return res;
    }
};
```
