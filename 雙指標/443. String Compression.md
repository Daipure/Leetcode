雙指針一個紀錄read 一個write

```cpp

class Solution {
public:


    int compress(vector<char>& chars) {
        // 超過十會分開存
        // 由前往後看是否有相同的
        // 使用雙指針
        // 一個read 一個write

        int i = 0;
        int write=0;
        int n = chars.size();

        while(i<n){
            char now = chars[i];
            int count = 1;
            int j=i+1;
            while(j<n){
                if(chars[j]==now){
                    count++;
                    j++;
                }
                else{
                    break;
                }
            }
            // write 紀錄該字母寫入的開始
            chars[write] = now;
            write++;

            if(count>1){
                string s = to_string(count);
                for(char c:s){
                    chars[write++]=c;
                }    
            }
            i=j;
        }

        return write;

    }
};
```
