使用滑動窗口
運用兩個變數來記錄兩個map是否相同

```cpp
class Solution {
public:
    string minWindow(string s, string t) {
        if(s.size()<t.size())return "";

        // 使用兩個map來記錄窗口的元素是否包含整個t
        // 當符合後縮小窗口

        unordered_map<char,int>window,countT;

        for(char c:t){
            countT[c]++;
        }

        // 使用兩個變數來輔助判斷窗口的元素是否包含整個t
        int have = 0 ,need = countT.size();
        int i = 0;
        int min_str = INT_MAX;
        pair<int,int>str_index={-1,-1};

        for(int j=0;j<s.size();j++){
            window[s[j]]++;
            // 該字元數量一樣種類+1
            if(countT[s[j]] && window[s[j]]==countT[s[j]]){
                have++;
            }
            // 如果種類一樣(window 可容納t)
            while( have == need){
                // 紀錄最短字串長度 以及他的l跟r座標

                if(j-i+1<min_str){
                    min_str = j-i+1;
                    str_index={i,j};
                }          

                // 窗口左移
                window[s[i]]--;

                // 要t有 windowt中小於t的數量才-1
                if(countT.count(s[i]) && window[s[i]]<countT[s[i]]){
                    have--;
                }
                
                i++;
            }    
        }
        if(min_str == INT_MAX)return "";
        return  s.substr(str_index.first,min_str);
    }
};

```
