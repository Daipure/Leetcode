兩個端點形成一個水柱
當要移動的時候因為寬度會變小 所以是移動較矮的
```cpp
class Solution {
public:
    int maxArea(vector<int>& height) {
        // 雙指針 計算左端點與右端點形成的面積
        int left = 0;
        int right = height.size()-1;

        int maxarea = 0;
        while(left <= right){
            // 寬乘高
            int h = min(height[left],height[right]);
            int w = right-left;
            int area = h*w;
            if(area>maxarea)maxarea=area;
            // 核心邏輯：移動較短邊的指針
            // 因為移動較長邊的指針，面積不可能會更大
            if(height[left]<height[right])left++;
            else right--;

        }
        return maxarea;

    }
};
```
