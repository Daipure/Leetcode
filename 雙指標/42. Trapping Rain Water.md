總面積為每格能接雨水的面積相加
每格能接雨水的面積為min(左隔板,右隔板) 減去自身高度

```cpp

class Solution {
public:
    int trap(vector<int>& height) {
        // 使用 prefix 和 suffix 兩個陣列 計算到i為止最大的高度
        // 因為每格能接得高度 min(左隔板,右隔板) 減去自身高度

        int n = height.size();
        vector<int>prefix(n,0);
        vector<int>suffix(n,0);

        prefix[0]=height[0];
        for(int i=1;i<n;i++){
            prefix[i]= max(prefix[i-1],height[i]);
        }
        suffix[n-1]=height[n-1];
        for(int i=n-2;i>=0;i--){
            suffix[i]= max(suffix[i+1],height[i]);
        }

        int res = 0;
        for(int i=0;i<n;i++){
            res += min(prefix[i],suffix[i])-height[i];
        }
        return res;
    }
};
```
