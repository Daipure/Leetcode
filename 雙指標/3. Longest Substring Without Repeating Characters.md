因為最長的子字串會連載一起 所以可以用 slide window 搭配雙指標
檢查有沒有重複 可以使用set
```cpp
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int n = s.size();
        if(n<=1)return n;

        // 因為最長的子字串會連載一起 所以可以用 slide window 搭配雙指標
        // 檢查有沒有重複 可以使用set
        int i=0;
        int max_len=0;

        unordered_set<int>set;
        for(int j=0;j<n;j++){
            //有重複
            while(set.find(s[j])!=set.end()){
                set.erase(s[i]); 
                i++;
            }
            set.insert(s[j]);
            max_len = max(max_len,(int)set.size());
        }
        return max_len;
        
    }
};
```
