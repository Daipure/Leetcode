給植物澆水
如果沒水了立刻補滿 並且澆水
所以每回合都該left++ right--
如果left == right 代表植物是基數
水多的交

```cpp
class Solution {
public:
    int minimumRefill(vector<int>& plants, int capacityA, int capacityB) {
        // 如果同個植物 水多的負責 水一樣alice 負責
        int left = 0;
        int right = plants.size()-1;

        int times = 0;
        int alice = capacityA;
        int bob  = capacityB;

        while(left<right){
            if(alice < plants[left]){
                alice = capacityA;
                times++;
            }
            alice-=plants[left];
            left++;

            if(bob < plants[right]){
                bob = capacityB;
                times++;
            }
            bob -= plants[right];
            right--;
        }
        if(left==right){
            if(max(alice,bob)<plants[left]){
                times++;
            }
        }
        return times;
    }
};
```
