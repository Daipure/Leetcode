判斷回文可以從頭和從中間
在找出最長回文中 從中間找速度更快

```cpp
class Solution {
public:
    string longestPalindrome(string s) {
        // 前面跟後面一樣
        // 從頭 O(n^3)
        // 改成使用中心點的只要O(n^2)

        // 分為基數和偶數探討
        
        int reslen = 0;
        int res_index = 0;

        for(int i=0;i<s.size();i++){
            int l=i,r=i;
            while(l>=0 && r<s.size() && s[l]==s[r]){
                if(r-l+1>reslen){
                    reslen = r-l+1;
                    res_index = l;
                }
                l--;
                r++;
            }
        }

        // 偶數
        // 中心點是兩數之間
        for(int i=0;i<s.size();i++){
            int l=i,r=i+1;
            while(l>=0 && r<s.size() && s[l]==s[r]){
                if(r-l+1>reslen){
                    reslen = r-l+1;
                    res_index = l;
                }
                l--;
                r++;
            }
        }

        return s.substr(res_index,reslen);
    }
};
```
