判斷有沒有cycle 想到快慢指針是否會相遇

```cpp

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    bool hasCycle(ListNode *head) {
        // 使用快慢指標 fast 碰到 slow 代表有cycle

        ListNode * slow = head;
        ListNode * fast = head;

        while(fast && fast->next){

            fast = fast->next->next;
            slow = slow->next;

            if(slow == fast)return true;
        }
        return false;
    }
};

```
