  // 分兩種情況
  // 第一間偷了 最後一間和第二間就不能偷
  // 第一間沒偷 最後一間可以偷

```cpp
class Solution {
public:
    int rob1(vector<int>& nums){
        int n =nums.size();
        if(n==0)return 0;
        if(n==1)return nums[0];

        vector<int>dp(n,0);
        // dp[n] = max(dp[n-1],dp[n-2]+nums[i])
        dp[0]=nums[0];
        dp[1]=max(nums[0],nums[1]);

        for(int i=2;i<n;i++){
            dp[i] = max(dp[i-1],dp[i-2]+nums[i]);
        }
        return dp[n-1];
    }

    int rob(vector<int>& nums) {
        // 分兩種情況
        // 第一間偷了 最後一間和第二間就不能偷
        // 第一間沒偷 最後一間可以偷

        int n = nums.size();
        if(n==1)return nums[0];

        // 不考慮第一間
        vector<int>nums1(nums.begin()+1,nums.end());
        // 不考慮最後一間
        vector<int>nums2(nums.begin(),nums.end()-1);
        return max(rob1(nums1),rob1(nums2));
    }
};
```
