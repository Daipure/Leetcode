刪除nums[i] 所以等於 nums[i]-1 和 nums[i]+1 都會被刪除
將題目轉換為打家劫舍問題
nums[i], nums[i]-1 和 nums[i]+1不能拿

```cpp

class Solution {
public:
    int deleteAndEarn(vector<int>& nums) {
        // 刪除nums[i] 所以等於 nums[i]-1 和 nums[i]+1 都會被刪除

        // 轉換為打家劫舍問題當你拿了 nums[i], nums[i]-1 和 nums[i]+1不能拿
        
        int max_val = *max_element(nums.begin(),nums.end());

        vector<int>a(max_val+1,0);
        for(int i=0;i<nums.size();i++){
            a[nums[i]] += nums[i];
        }

        vector<int>dp(max_val+1,0);
        dp[0] = a[0];
        dp[1] = max(a[0],a[1]);

        for(int i=2;i<max_val+1;i++){
            dp[i] = max(dp[i-1],dp[i-2]+a[i]);
        }

        return dp[max_val];
        
    }
};
```
