構建長度i字串的方法為 dp[i-zero] + [i-one]


```cpp


class Solution {
public:
    int countGoodStrings(int low, int high, int zero, int one) {
        // 滿足長度為 low 到 high 長度的字串
        // 由zero 個0 和 one個1 所組成

        vector<int>dp(high+1,0);
        int MOD = 1000000007;

        // 構造空字串    
        dp[0] = 1;

        for(int i=0;i<=high;i++){
            if(i-one>=0){
                dp[i] = (dp[i]+dp[i-one])%MOD;
            }
            if(i-zero>=0){
                dp[i] = (dp[i]+dp[i-zero])%MOD;
            }           
        }

        int ans =0;
        for(int i=low;i<=high;i++){
            ans= (ans+dp[i])%MOD;
        }
        return ans;
    }
};
```
