二分搜尋 判斷應該捨去哪一半 如果中間節點大於左邊 代表是遞增 因此去尋找右邊
區間要遵循左庇右閉 或左閉右開


```cpp

class Solution {
public:
    int findMin(vector<int>& nums) {
        // 使用二分搜尋法
        // 判斷應該捨去哪一半
        // 如果中間節點大於左邊 代表是遞增 因此去尋找右邊

        int l=0;
        int r=nums.size()-1;
        int min_val = nums[0];

        // 遵守左閉又閉

        while(l<=r){
            if(nums[l]<=nums[r]){
                min_val = min(min_val,nums[l]);
            }
            int m = (l+r)/2;
            min_val = min(min_val,nums[m]);

            // 如果中間節點大於左邊 代表是遞增 因此去尋找右邊
            if(nums[m]>=nums[l]){
                l = m+1;
            }
            // 找左邊
            else {
                r = m-1;
            }

        }

        return min_val;
    }
};

```
