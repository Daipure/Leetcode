// 使用map 陣列紀錄每個字母的次數 再比次數

```cpp

class Solution {
public:
    int maxFreqSum(string s) {
        unordered_set<char>vowel ={'a','e','i','o','u'};
        int vow_count = 0,con_count=0;

        unordered_map<char,int>map;

        for(char c:s){
            map[c]++;
        }

        // 子音比大小
        for(auto it:map){
            if( vowel.count(it.first)==0){
                con_count = max(con_count,it.second);
            }
            else{
                vow_count = max(vow_count,it.second);
            }
        }

        return con_count+vow_count;
    }
};

```
