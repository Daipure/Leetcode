找出最長的連續 在不使用排序的前提下 使用查表
因為要查找是否是連續的所以使用set往前或往後尋找
為了避免每個都往前或往後尋找
只在當這個數是第一個(沒有當前數-1)的時候往後尋找

```cpp
class Solution {
public:
    int longestConsecutive(vector<int>& nums) {
        if(nums.size()<2)return nums.size();
        // 因為要查找是否是連續的所以使用set往前或往後尋找
        // 為了避免每個都往前或往後尋找
        // 只在當這個數是第一個(沒有當前數-1)的時候往後尋找

        unordered_set<int>set(nums.begin(),nums.end());
        int max_len = 1;
        int cur_len = 1;

        for(int num:set){
            // 只在當這個數是第一個(沒有當前數-1)的時候往後尋找
            if(set.find(num-1)!=set.end())continue;
            cur_len = 1;
            while(set.find(num+1)!=set.end()){
                cur_len++;
                num+=1;
                
            }
            if(cur_len>max_len)max_len = cur_len;
        }
        return max_len;
    }
};
```
