當遇到一樣的分組時 可以使用map
將key一樣的放在一起 當key相同 value可以使用容器 push_back進去
這題的key藉由Anagrams特性 將字串排列當作key
也可以記數組成字串當作key

```cpp
class Solution {
public:
    vector<vector<string>> groupAnagrams(vector<string>& strs) {
        vector<vector<string>>result;
        unordered_map<string,vector<string>>map;
        // 因為是Anagrams所以排序後都一樣 代表可以當作key
        for(string s:strs){
            string key = s;
            sort(key.begin(),key.end());
            map[key].push_back(s);
        }

        for(auto it:map){
            result.push_back(it.second);
        }
        return result;

    }
```
