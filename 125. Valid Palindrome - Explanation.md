class Solution {
public:
    bool isPalindrome(string s) {
        // 去掉特殊字 轉小寫
        // 看反轉字串看是否相同
        string new_str = "";
        for(char c:s){
            if(isalnum(c)){
                new_str += tolower(c);
            }
        }
        return new_str == string(new_str.rbegin(),new_str.rend());
    }
};
