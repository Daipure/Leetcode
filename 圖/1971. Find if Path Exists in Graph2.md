並查集 join find 函式

```cpp

class Solution {
public:
    vector<int> father;

    // 初始化，讓每個節點自成一個集合
    void init(int n) {
        father.resize(n); // 先調整 vector 大小
        for (int i = 0; i < n; i++) {
            father[i] = i;
        }
    }

    // 查詢根節點 + 路徑壓縮優化
    int find(int u) {
        if (u == father[u]) return u;
        return father[u] = find(father[u]); // 路徑壓縮
    }

    // 合併兩個節點所在的集合
    void join(int u, int v) {
        int rootU = find(u);
        int rootV = find(v);
        if (rootU != rootV) {
            father[rootU] = rootV;
        }
    }

    bool validPath(int n, vector<vector<int>>& edges, int source, int destination) {
        // 修正點 1: 在所有操作前，必須先初始化並查集
        init(n);

        // 遍歷所有邊，進行合併操作
        for (const auto& edge : edges) {
            join(edge[0], edge[1]);
        }

        // 最終判斷 source 和 destination 的根節點是否相同
        return find(source) == find(destination);
    }
};
```
