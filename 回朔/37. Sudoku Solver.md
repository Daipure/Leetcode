解數獨 全部順利填完或是填不了 return 狀態
每格 都用1到9 去測試

```cpp
class Solution {
public:
    bool is_valid(vector<vector<char>>& board,int x,int y,char num){
        for(int i=0;i<9;i++){
            // 直行
            if(board[i][y]==num)return false;
            // 列
            if(board[x][i]==num)return false;
        }
        int row_index = x/3*3;
        int col_index = y/3*3;
        for(int i = row_index ; i<row_index+3;i++){
            for(int j = col_index ; j<col_index+3;j++){
                if(board[i][j]==num)return false;
            }
        }
        return true;
    }
    bool backtracking(vector<vector<char>>& board){
        for(int i=0;i<9;i++){
            for(int j=0;j<9;j++){
                if(board[i][j]!='.')continue;
                for(int num=1;num<=9;num++){
                    // 還沒填且合法
                    if(is_valid(board,i,j,num+'0')){
                        board[i][j]=num+'0';
                        if(backtracking(board))return true;
                        board[i][j]='.';
                    }   
                }
                return false;
            }
        }
        return true;
    }
    void solveSudoku(vector<vector<char>>& board) {
        backtracking(board);
    }
};
```
