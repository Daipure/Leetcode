bit 反轉 利用左移以及右移

```cpp

class Solution {
public:
    int reverseBits(int n) {
        // n最後一位 n&1 
        // 新的數  0 

        int result = 0;
        for(int i=0;i<32;i++){
            // n最後一位 n&1 
            int lsb = n&1;

            //result左移空出位置
            result = result<<1;

            // or lsb 等價於 +lsb
            result = result | lsb;
            
            // n右移
            n=n>>1;
        }
        return result;

        
    }
};
```
