利用xor 依樣會消掉的特性

```cpp

class Solution {
public:
    int missingNumber(vector<int>& nums) {

        // 利用xor 一樣的數會消掉的特性
        // 將 1到n 與 nums 做xor 

        int n = nums.size();
        int result = n;

        for(int i=0;i<n;i++){
            result = result ^ i ^ nums[i];
        }
        return result;
    }
};
```
