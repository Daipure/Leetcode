可以透過%2取出最右邊的值是1和0
也可以透過 n & (n-1)

```cpp

class Solution {
public:
    int hammingWeight(int n) {

        //  bit 右移
        // int res = 0;

        // while(n){
        //     res += n%2;
        //     n=n>>1;
        // }
        // return res;

        //hammingWeight
        int res = 0;
        while(n){
            n = n&n-1;
            res++;
        }
        return res;
    }
};
```
