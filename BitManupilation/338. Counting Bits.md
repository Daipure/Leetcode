計算每個數有幾個1 可以發現 dp[i] = dp[i>>1]+ i&1
例如 11是1011 = 右移1位+%2 或是 i%2 取最右邊的那個數的結果

class Solution {
public:
    vector<int> countBits(int n) {
        vector<int>dp(n+1,0);
        // dp 算法

        dp[0]=0;

        for(int i=1;i<=n;i++){
            // 例如 11是1011 = 右移1位+%2 或是 i%2 取最右邊的那個數的結果 
            dp[i]= dp[i>>1] + (i&1);
        }
        return dp;
    }
};
