用一個值紀錄路徑中的最大節點

```cpp

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    
    
    int count = 0;

    void dfs(TreeNode* root, int last){
        if(root==NULL)return;
        if(root->val >= last){
            count++;
            last = root->val;
        }
        dfs(root->left,last);
        dfs(root->right,last);
    }

    int goodNodes(TreeNode* root) {
        dfs(root,-10000);
        return count;
    }


};
```
