先將每個點 的座標以及值 使用map記錄下來
然後根據順序去存答案

```cpp

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:

    // 需要有每個點的座標 可以透過dfs計算每個點的座標 並加入到map 中
    map<int,vector<pair<int,int>>>group;

    void dfs(TreeNode* root,int x , int y){
        if(root==NULL)return;
        group[y].push_back({x,root->val});
        dfs(root->left,x+1,y-1);
        dfs(root->right,x+1,y+1);
    }

    vector<vector<int>> verticalTraversal(TreeNode* root) {
        // 根據column 收割
        dfs(root,0,0);

        // 根據row排序

        vector<vector<int>>res;

        // 遍歷 map 中的每一個 key-value pair
        for(auto& vec : group){ 
            // 對 value (也就是 vector) 進行排序
            sort(vec.second.begin(), vec.second.end()); 

            vector<int> temp;
            // 遍歷 value (也就是 vector)
            for(auto& p : vec.second){ 
                temp.push_back(p.second);
            }
            res.push_back(temp);
        }
        return res; // 回傳最終結果

    }

};
```
