使用dp計算不同的三角型

```cpp
class Solution {
public:
    int minScoreTriangulation(vector<int>& values) {
        // 會產生 n-2 個三角形
        // 可以切割成子問題
        // 多邊形最簡單的方法就是用起始頂點 i 和結束頂點 j 來定義。
        // 為了進行三角剖分，我們必須選擇一個中間點 k (其中 i < k < j)，來和 i、j 形成一個三角形 (i, k, j)。     
        vector<vector<int>>dp(50,vector<int>(50,INT_MAX));
        int n = values.size();

        // 相鄰兩個點 構不成三角形 初始化為0
        for (int i = 0; i < n - 1; ++i) {
            dp[i][i+1] = 0;
        }

        for (int len = 2; len < n; ++len) {
            // 2. 中層迴圈：遍歷所有可能的起始點 i
            for (int i = 0; i <= n - 1 - len; ++i) {
                int j = i + len; // 計算終點 j

                // 3. 內層迴圈：在 [i, j] 區間內尋找最佳分割點 k
                for (int k = i + 1; k < j; ++k) {
                    int current_score = dp[i][k] + dp[k][j] + values[i] * values[k] * values[j];
                    dp[i][j] = min(dp[i][j], current_score);
                }
            }
        }

        return dp[0][n-1];
    }
};

```
