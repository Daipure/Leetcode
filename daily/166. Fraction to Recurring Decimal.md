使用長除法 當餘數重複時代表是循環小數
利用map紀錄餘數的位置

```cpp
class Solution {
public:
    string fractionToDecimal(int numerator, int denominator) {
        // 分子為0
        if (numerator == 0) return "0";

        string res = "";
        // 負數
        if ((numerator > 0 && denominator < 0) || (numerator < 0 && denominator > 0)) { 
            res += "-";
        }

        long long long_numerator = abs((long long)numerator);
        long long long_denominator = abs((long long)denominator);
        res += to_string(long_numerator / long_denominator);
        long long remainder = long_numerator % long_denominator;
        // 整除
        if (remainder == 0) {
            return res;
        }

        // 不整除
        res += ".";
        unordered_map<long long, int> map;
        while (remainder != 0) {
            if (map.count(remainder) != 0) {
                // 循環
                res.insert(map[remainder], "(");
                res += ")";
                break;
            }
            map[remainder] = res.length();
            remainder *= 10;
            res += to_string(remainder / long_denominator);
            remainder = remainder % long_denominator;
        }
        return res;
    }
};

```
