解析字串得到位置的值

```cpp

class Spreadsheet {
public:

    vector<vector<int>> grid{1001, vector<int>(26, 0)};
    Spreadsheet(int rows) {}
    pair<int,int> parse_cell(string cell){
        int col = cell[0]-'A';
        int row = stoi(cell.substr(1))-1;
        return {row,col};
    }

    void setCell(string cell, int value) {
        auto pair = parse_cell(cell);
        int row = pair.first;
        int col = pair.second;
        grid[row][col]=value;
    }
    
    void resetCell(string cell) {
        auto pair = parse_cell(cell);
        int row = pair.first;
        int col = pair.second;
        grid[row][col]=0;        
    }
    
    int getValue(string formula) {
        formula = formula.substr(1);
        int pos = formula.find('+');
        string a = formula.substr(0,pos);
        string b = formula.substr(pos+1);
        int num1=0,num2=0;
        if(isalpha(a[0])){
            auto pair = parse_cell(a);
            num1 = grid[pair.first][pair.second];
        }
        else{
            num1 = stoi(a);
        }
        if(isalpha(b[0])){
            auto pair = parse_cell(b);
            num2 = grid[pair.first][pair.second];
        }
        else{
            num2 = stoi(b);
        }
        return num1+num2;
    }
};

/**
 * Your Spreadsheet object will be instantiated and called as such:
 * Spreadsheet* obj = new Spreadsheet(rows);
 * obj->setCell(cell,value);
 * obj->resetCell(cell);
 * int param_3 = obj->getValue(formula);
 */
```
