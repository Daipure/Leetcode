n = a+b  找出一組解使得a+b=n 且a,b都不含0 暴力解判斷


```cpp
class Solution {
public:

    bool is_valid(string &s){
        for(char c:s){
            if(c=='0')return false;
        }
        return true;
    }
    vector<int> getNoZeroIntegers(int n) {
        for(int a=1;a<n;a++){
            int b = n-a;

            string sa = to_string(a);
            string sb = to_string(b);
            
            if(is_valid(sa)&&is_valid(sb)){
                return {a,b};
            }
        }
        return {0,0};
    }
};

```
