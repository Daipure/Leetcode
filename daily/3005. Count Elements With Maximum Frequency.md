紀錄每個值的頻率 和最大頻率
如果是最大頻率就加總


```cpp

class Solution {
public:
    int maxFrequencyElements(vector<int>& nums) {
        vector<int>vec(105,0);
        int max = 0;

        for(int i=0;i<nums.size();i++){
            vec[nums[i]]+=1;
            if(vec[nums[i]]>=max){
                max = vec[nums[i]];
            }
        }

        int count = 0;
        for(int i=0;i<vec.size();i++){
            if(vec[i]==max){
                count+=max;
            }
        }
        return count;
    }
};
```
