運用座標排序使得 固定i的情況下 j 一定在右邊 當j的座標小於i 且j的高比之前出現的j的高都還要大 代表沒重疊


```cpp

class Solution {
public:
    int numberOfPairs(vector<vector<int>>& points) {
        int n = points.size();
        //  x座標升序排列 y座標降序
        sort(points.begin(),points.end(),[](vector<int>&a ,vector<int>&b){
            if(a[0]==b[0]){
                return a[1]>b[1];
            }
            return a[0]<b[0];
        });

        int count = 0;
        for(int i=0;i<n;i++){
            int right_botten = INT_MIN;
            for(int j=i+1;j<n;j++){
                // j的y要比i的y小 且j要大於上一個y 不然會重疊
                if(points[j][1]<=points[i][1] && points[j][1]>right_botten){
                    count++;
                    right_botten=points[j][1];
                }
            }
        }
        return count;
    }
};

```
