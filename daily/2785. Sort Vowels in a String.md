只需排序母音 所以如果是母音就抓出來排序

```cpp

class Solution {
public:
    string sortVowels(string s) {
        // 找出母音的 排序後放回

        unordered_set<char> vowel_set = {'a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'};
        string vowels="";
        for(char c:s){
            if(vowel_set.find(c)!=vowel_set.end()){
                vowels+=c;
            }
        }

        sort(vowels.begin(),vowels.end());

        
        int index = 0;
        for(int i=0;i<s.size();i++){
            // 母音從小到大放回
            if(vowel_set.find(s[i])!=vowel_set.end()){
                s[i]= vowels[index];
                index++;      
            }
        }
        return s;
    }
};

```
