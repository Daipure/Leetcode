九宮格找到該格最起始的值 i/3*3 j/3*3
找到是第幾個九宮格 i/3*3 + j/3
本題使用查表 建立橫排 直排 九宮格三個表 使用[該牌][num]的方式快速查 如果true代表有出現過 即return false

class Solution {
public:
    bool isValidSudoku(vector<vector<char>>& board) {
        // 使用查表判斷是否有出現過
        bool row[9][10]={false};
        bool col[9][10]={false};
        bool box[9][10]={false};
        
        for(int i=0;i<9;i++){
            for(int j=0;j<9;j++){
                if(board[i][j]!='.'){
                    int num = board[i][j]-'0';
                    // 算在第幾個九宮格
                    int b_index = (i/3)*3 +(j/3);
                    // 使用查表判斷是否有出現過
                    if(row[i][num] || col[j][num] || box[b_index][num]){
                        return false;
                    }
                    row[i][num]=true;
                    col[j][num]=true;
                    box[b_index][num]=true;
                }
            }
        }
        return true;
    }
};
