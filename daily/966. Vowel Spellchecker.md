修正器根據3個規則優先順序來修正 1.完美 2.大小寫 3母音錯誤
如果每個詞都在不同規則上用雙層迴圈去跑，當 wordlist 和 queries 很大時，會非常慢，在 LeetCode 上會「超時 (Time Limit Exceeded)」。
所以使用查表的方式
```cpp

class Solution {
public:
    // 輔助函式：將字串轉為小寫
    string toLower(string& s) {
        string lower_s = s;
        transform(lower_s.begin(), lower_s.end(), lower_s.begin(),
                  [](unsigned char c){ return tolower(c); });
        return lower_s;
    }

    // 輔助函式：將字串去母音 (母音轉為 '#' 方便比對)
    string devowel(string& s) {
        string lower_s = toLower(s);
        for (char& c : lower_s) {
            if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u') {
                c = '#';
            }
        }
        return lower_s;
    }

    vector<string> spellchecker(vector<string>& wordlist, vector<string>& queries) {
        // --- 步驟一：預處理 wordlist，建立三個查詢表 ---

        // 1. 用於規則一：完全符合
        unordered_set<string> perfect_match(wordlist.begin(), wordlist.end());

        // 2. 用於規則二：忽略大小寫
        unordered_map<string, string> cap_map;
        
        // 3. 用於規則三：忽略母音
        unordered_map<string, string> vowel_map;

        for (string& word : wordlist) {
            string lower_word = toLower(word);
            string devowel_word = devowel(word);

            // 如果 key 不存在才放入，確保只儲存「第一個」符合的單字
            if (cap_map.find(lower_word) == cap_map.end()) {
                cap_map[lower_word] = word;
            }
            if (vowel_map.find(devowel_word) == vowel_map.end()) {
                vowel_map[devowel_word] = word;
            }
        }

        // --- 步驟二：遍歷 queries，按順序查詢 ---
        vector<string> res;
        for (string& query : queries) {
            // 規則 1: 完全符合
            if (perfect_match.count(query)) {
                res.push_back(query);
                continue; // 找到答案，處理下一個 query
            }

            // 規則 2: 忽略大小寫
            string lower_query = toLower(query);
            if (cap_map.count(lower_query)) {
                res.push_back(cap_map[lower_query]);
                continue; // 找到答案，處理下一個 query
            }

            // 規則 3: 忽略母音
            string devowel_query = devowel(query);
            if (vowel_map.count(devowel_query)) {
                res.push_back(vowel_map[devowel_query]);
                continue; // 找到答案，處理下一個 query
            }

            // 都沒找到
            res.push_back("");
        }

        return res;
    }
};

```
