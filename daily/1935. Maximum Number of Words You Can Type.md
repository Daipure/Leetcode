可以邊遍歷邊檢查
如果遇到空白或結尾代表一個詞的結束 看到flag有沒有變true
使用set 建立brokenLetters字典
檢查text 中的word是否包含brokenLetters

```cpp

class Solution {
public:
    int canBeTypedWords(string text, string brokenLetters) {
        // 使用set 建立brokenLetters字典
        // 檢查text 中的word是否包含brokenLetters

        int count = 0;
        unordered_set<char>set(brokenLetters.begin(),brokenLetters.end());
        int n = text.size();
        bool flag = false;

        for(int i=0;i<n;i++){
            char c = text[i];
            if(c==' '){
                if(!flag){
                    count++;
                }
                flag = false;
            }
            else if(!flag){
                if(set.count(c)!=0){
                    flag = true;
                }
            }
        }

        if(!flag)count++;

        return count;
    }
};

```
