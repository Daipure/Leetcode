因為題目要一直去計算 並動態更新 所以使用priority que 要注意兩個整數除法答案會是整數 因此要用幅點數



```cpp
class Solution {
public:
    double maxAverageRatio(vector<vector<int>>& classes, int extraStudents) {
        // 每次找出最大增量的
        // 使用最大堆 1.會重複計算更新後的classes的最大增量
        priority_queue<pair<double,int>>pq;
        // 初始化pq
        for(int i=0;i<classes.size();i++){
            double r = double(classes[i][0]+1)/double(classes[i][1]+1)-double(classes[i][0])/double(classes[i][1]); 
            pq.push({r,i});
        }

        while(extraStudents--){
            double max_incre = pq.top().first;
            int i = pq.top().second;
            pq.pop();
            // 將更新後的加進去pq
            // 將他更新值 並計算增量
            classes[i][0]+=1;
            classes[i][1]+=1;
            double r = double(classes[i][0]+1)/double(classes[i][1]+1)-double(classes[i][0])/double(classes[i][1]); 
            pq.push({r,i});
        }

        double res=0;
        for(int i=0;i<classes.size();i++){
            res += double(classes[i][0])/double(classes[i][1]);
        }
        return res/classes.size();
    }
};
```
