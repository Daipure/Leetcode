class Solution {
public:

    int gcd(int a,int b){
        while(b!=0){
            int r = a%b;
            a=b;
            b=r;
        }
        return a;
    }

    long long  lcm(int a,int b){
        if(a==0 || b==0)return 0;
        return  (long long)a*b/(gcd(a,b));
    }
    vector<int> replaceNonCoprimes(vector<int>& nums) {


        vector<int>sta;
        for(int i=0;i<nums.size();i++){
            sta.push_back(nums[i]);
            while(sta.size()>=2){
                int a = sta.back();
                sta.pop_back();
                int b = sta.back();
        
                if(gcd(a,b)!=1){
                    sta.back()=lcm(a,b);
                }
                else{
                    sta.push_back(a);
                    break;
                }

            }
        }
        return sta;
    }
};
