動態規劃題目
兩個嚴格遞增的 也可能發生在一個連續遞增的陣列中
所以要分開討論
1. 將連續遞增的分兩半
2. 兩個相鄰的 i-dp[i]為前一個的結束座標

```cpp
class Solution {
public:
    int maxIncreasingSubarrays(vector<int>& nums) {
        int n = nums.size();
        
        vector<int>dp(n,1);

        for(int i=1;i<n;i++){
            if(nums[i]>nums[i-1]){
                dp[i] = dp[i-1]+1;
            }
        }

        int maxL = 0;
        for(int i=0;i<n;i++){
            maxL = max(maxL, dp[i] / 2);
            // 前一個結束的
            int pre = i - dp[i]; 
            if (pre >= 0) {
                maxL = max(maxL, min(dp[i], dp[pre]));
            }
        }
        return maxL ;
    }
};

```
