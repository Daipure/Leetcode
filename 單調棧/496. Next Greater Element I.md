使用單調棧找出nums中第一個比較大的
透過映射關係到nums1
使用map 去知道nums2的值對應到nums1的座標

```cpp

class Solution {
public:
    vector<int> nextGreaterElement(vector<int>& nums1, vector<int>& nums2) {
        // 使用單調棧找出nums中第一個比較大的
        // 透過映射關係到nums1
        // 使用map 去知道nums2的值對應到nums1的座標
        if(nums1.size()==0)return {};

        unordered_map<int,int>umap;

        for(int i=0;i<nums1.size();i++){
            umap[nums1[i]]=i;
        }

        stack<int>sta;
        sta.push(0);
        vector<int>res(nums1.size(),-1);

        // nums2[i] <= nums2[sta.top] push進去
        
        for(int i=1;i<nums2.size();i++){
            if(nums2[i]<=nums2[sta.top()]){
                sta.push(i);
            }
            else{
                // 找到大於的
                while(!sta.empty() && nums2[i]>nums2[sta.top()]){
                    // sta.top 索引
                    // 透過map找到在nums1 存在的位置更新值
                    if(umap.count(nums2[sta.top()])!=0){
                        res[umap[nums2[sta.top()]]]=nums2[i];
                    }
                    sta.pop();
                }
                sta.push(i);
            }
        }
        return res;
    }
};

```
