stringstream 將字串轉為字符串流  stringstream ss(string)
getline(字符串流,儲存的字串,'分割字符')
getline(ss,str,'.')
數量不夠就補0 透過判斷長度的方式
 
``` cpp

class Solution {
public:
    
    vector<int>split(string &version){
        stringstream ss(version);
        string  str;
        vector<int>vec;
        while(getline(ss,str,'.')){
            vec.push_back(stoi(str));
        }
        return vec;
    }



    int compareVersion(string version1, string version2) {
        
        vector<int>vec1 = split(version1);
        vector<int>vec2 = split(version2);

        int n1 = vec1.size();
        int n2 = vec2.size();

        int i=0;
        // 如果size 不夠就補0
        while(i<n1 || i<n2){
            int num1 = i>=n1?0:vec1[i];
            int num2 = i>=n2?0:vec2[i];
            if(num1>num2)return 1;
            if(num1<num2)return -1;
            i++;
        }
        return 0;
    }
};
```
