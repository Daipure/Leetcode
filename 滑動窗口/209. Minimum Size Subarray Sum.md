由於這題的單調性(右指針往右值會變大，左指針往右值會縮小)
可以使用滑動窗口來找尋最小的陣列長度
  
```cpp
class Solution {
public:
    int minSubArrayLen(int target, vector<int>& nums) {
        // 雙指針

        int left = 0;
        int sum = 0;
        int res = INT_MAX;

        for(int right = 0;right<nums.size();right++){
            sum+=nums[right];

            while(sum>=target){
                res = min(res,right-left+1);
                sum-=nums[left];
                left++;
            }
        }
        return res == INT_MAX ? 0 : res;
    }
};

```
