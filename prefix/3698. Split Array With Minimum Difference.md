切割陣列使得找到一個切割點 左邊是遞增陣列 右邊是遞減 絕對值相減最小
使用prefix陣列 去得到目前加總的值
全部 減去右邊相加 減去左邊相加
絕對值 為 totalsum  - prefix[left] - prefix[left]
使用incre decre vector 紀錄到該值是否為遞增陣列
```cpp

class Solution {
public:
    long long splitArray(vector<int>& nums) {
        // 左邊遞增 右邊遞減
        long long res = LLONG_MAX;
        int n = nums.size();
        if(n<2)return -1;

        // 前綴和計算
        vector<long long>prefix(n,0);
        prefix[0] = nums[0]; // 先設定好第一個元素的初始值
        for (int i = 1; i < n; i++) {
            prefix[i] = prefix[i - 1] + nums[i]; // 從第二個元素開始累加
        }
        long long total = prefix[n-1];
    
        vector<bool>incre(n,false);
        vector<bool>decre(n,false);

        incre[0]=true;
        decre[n-1]=true;
        // 判斷遞增
        for(int i=1;i<nums.size();i++){
            if(incre[i-1]==false)break;
            if(nums[i]>nums[i-1] && incre[i-1]==true){
                incre[i]=true;
                
            }
        }   
        // 判斷遞減
        for(int i=n-2;i>=0;i--){
            if(decre[i+1]==false)break;
            if(nums[i]>nums[i+1] && decre[i+1]==true){
                decre[i]=true;
                
            }
        }

        // 找切割點
        for(int i=0;i<n-1;i++){
            if( incre[i]==true && decre[i+1]==true){
                res = min(res, abs(total-prefix[i]-prefix[i]));
            }
        }
        return res==LLONG_MAX? -1:res;
        
    }
};

```
